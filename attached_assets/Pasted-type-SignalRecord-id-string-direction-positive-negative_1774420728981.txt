type SignalRecord = {
  id: string;
  direction?: "positive" | "negative" | "neutral";
  strengthScore?: number;
  reliabilityScore?: number;
  independenceScore?: number;
};

type RecalcResult = {
  score: number;
  calculatedAt: string;
  signalCount: number;
};

async function getSignalsForCase(caseId: string): Promise<SignalRecord[]> {
  // Replace with your DB logic
  const response = await fetch(`http://localhost:3000/internal-db/cases/${caseId}/signals`);
  if (!response.ok) {
    throw new Error(`Could not load signals for case ${caseId}`);
  }
  return response.json();
}

async function saveCaseScore(caseId: string, score: number, calculatedAt: string) {
  // Replace with your DB logic
  const response = await fetch(`http://localhost:3000/internal-db/cases/${caseId}/score`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ score, calculatedAt }),
  });

  if (!response.ok) {
    throw new Error(`Could not save recalculated score for case ${caseId}`);
  }
}

function clamp01(value: number): number {
  return Math.max(0, Math.min(1, value));
}

function computeSimpleScore(signals: SignalRecord[]): number {
  if (!signals.length) return 0;

  let total = 0;

  for (const signal of signals) {
    const strength = clamp01(signal.strengthScore ?? 0.5);
    const reliability = clamp01(signal.reliabilityScore ?? 0.5);
    const independence = clamp01(signal.independenceScore ?? 0.5);

    const base = strength * reliability * independence;

    const directionMultiplier =
      signal.direction === "positive"
        ? 1
        : signal.direction === "negative"
        ? -1
        : 0;

    total += base * directionMultiplier;
  }

  // Convert to simple 0-100 display scale
  const normalized = ((total / Math.max(signals.length, 1)) + 1) / 2;
  return Math.round(clamp01(normalized) * 1000) / 10;
}

export async function recalculateCaseScore(caseId: string): Promise<RecalcResult> {
  const signals = await getSignalsForCase(caseId);
  const score = computeSimpleScore(signals);
  const calculatedAt = new Date().toISOString();

  await saveCaseScore(caseId, score, calculatedAt);

  return {
    score,
    calculatedAt,
    signalCount: signals.length,
  };
}
