function generateSignalSuggestions(context):

    signals = []

    if context.disease == "alopecia":

        signals.append("AAD guideline update")
        signals.append("Hair-loss treatment protocol change")
        signals.append("Insurance coverage policy change")

    if context.questionType == "comparative":

        signals.append("Difference in prescribing familiarity")
        signals.append("Difference in patient volume")

    return signals
