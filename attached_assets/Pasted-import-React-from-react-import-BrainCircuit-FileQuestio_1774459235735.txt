import React from "react";
import {
  BrainCircuit,
  FileQuestion,
  Sparkles,
  Radar,
  Radio,
  ClipboardCheck,
  LayoutDashboard,
  BookOpen,
  BarChart3,
  Map,
  AlertCircle,
  ChevronRight,
} from "lucide-react";
import { ActiveQuestion, NavKey } from "../types/navigation";

type Props = {
  activeItem: NavKey;
  onNavigate: (item: NavKey) => void;
  activeQuestion: ActiveQuestion;
};

type Item = {
  key: NavKey;
  label: string;
  description: string;
  icon: React.ReactNode;
  isChild?: boolean;
  disabled?: boolean;
};

function cn(...classes: Array<string | false | null | undefined>) {
  return classes.filter(Boolean).join(" ");
}

function SidebarButton({
  item,
  activeItem,
  onNavigate,
}: {
  item: Item;
  activeItem: NavKey;
  onNavigate: (item: NavKey) => void;
}) {
  const active = activeItem === item.key;

  return (
    <button
      type="button"
      disabled={item.disabled}
      onClick={() => !item.disabled && onNavigate(item.key)}
      className={cn(
        "w-full rounded-2xl border text-left transition-all",
        item.isChild ? "pl-10 pr-3 py-2.5" : "px-3 py-3",
        active
          ? "border-blue-700 bg-blue-950 text-white"
          : "border-transparent text-slate-300 hover:bg-white/5 hover:text-white",
        item.disabled ? "cursor-not-allowed opacity-50" : "cursor-pointer"
      )}
    >
      <div className="flex items-start gap-3">
        <div className={cn("mt-0.5 shrink-0", item.isChild && "ml-3")}>{item.icon}</div>
        <div className="min-w-0 flex-1">
          <div className="flex items-center justify-between gap-3">
            <div className="truncate text-[15px] font-medium">{item.label}</div>
            {active && <ChevronRight className="h-4 w-4 shrink-0 text-slate-300" />}
          </div>
          <div className="mt-1 text-[11px] text-slate-400">{item.description}</div>
        </div>
      </div>
    </button>
  );
}

function SectionTitle({ children }: { children: React.ReactNode }) {
  return (
    <div className="px-3 pt-6 pb-2 text-xs font-semibold uppercase tracking-[0.18em] text-slate-500">
      {children}
    </div>
  );
}

export default function CiosSidebar({ activeItem, onNavigate, activeQuestion }: Props) {
  const decision: Item[] = [
    {
      key: "questions",
      label: "Questions",
      description: "Define what we are predicting",
      icon: <FileQuestion className="h-5 w-5" />,
    },
    {
      key: "adopter-discovery",
      label: "Adopter Discovery",
      description: activeQuestion
        ? "Define who will act on the active question"
        : "Select or create a question first",
      icon: <Sparkles className="h-4 w-4" />,
      isChild: true,
      disabled: !activeQuestion,
    },
  ];

  const evidence: Item[] = [
    {
      key: "event-radar",
      label: "Event Radar",
      description: "Track future signals",
      icon: <Radar className="h-5 w-5" />,
    },
    {
      key: "signal-detection",
      label: "Signal Detection",
      description: "Capture new information",
      icon: <Radio className="h-5 w-5" />,
    },
    {
      key: "signal-review",
      label: "Signal Validation",
      description: "Confirm signal quality",
      icon: <ClipboardCheck className="h-5 w-5" />,
    },
  ];

  const prediction: Item[] = [
    {
      key: "dashboard",
      label: "Dashboard",
      description: "Show current forecast",
      icon: <LayoutDashboard className="h-5 w-5" />,
    },
    {
      key: "forecast-ledger",
      label: "Forecast Ledger",
      description: "Track prediction history",
      icon: <BookOpen className="h-5 w-5" />,
    },
  ];

  const learning: Item[] = [
    {
      key: "calibration",
      label: "Calibration",
      description: "Measure accuracy",
      icon: <BarChart3 className="h-5 w-5" />,
    },
    {
      key: "system-map",
      label: "System Map",
      description: "Understand how the app works",
      icon: <Map className="h-5 w-5" />,
    },
  ];

  return (
    <aside className="flex h-screen w-[340px] flex-col border-r border-white/10 bg-[#071238] text-white">
      <div className="border-b border-white/10 px-6 py-5">
        <div className="flex items-start gap-4">
          <div className="flex h-14 w-14 items-center justify-center rounded-2xl border border-blue-700 bg-blue-950 text-blue-300">
            <BrainCircuit className="h-7 w-7" />
          </div>
          <div className="pt-1">
            <div className="text-[34px] font-semibold leading-none tracking-tight">CIOS</div>
            <div className="mt-1 text-sm text-slate-400">Strategic Intelligence Engine</div>
          </div>
        </div>

        <div className="mt-4 rounded-2xl border border-white/10 bg-white/5 p-4">
          <div className="mb-2 text-xs font-semibold uppercase tracking-[0.14em] text-slate-400">
            Active Question Context
          </div>
          {activeQuestion ? (
            <>
              <div className="text-sm font-medium text-white">{activeQuestion.title}</div>
              <div className="mt-3 flex flex-wrap gap-2">
                {activeQuestion.therapyArea && (
                  <span className="rounded-full bg-slate-800 px-2.5 py-1 text-[11px] text-slate-300">
                    {activeQuestion.therapyArea}
                  </span>
                )}
                {activeQuestion.targetPopulation && (
                  <span className="rounded-full bg-slate-800 px-2.5 py-1 text-[11px] text-slate-300">
                    {activeQuestion.targetPopulation}
                  </span>
                )}
              </div>
            </>
          ) : (
            <div className="flex items-center gap-2 text-sm text-amber-300">
              <AlertCircle className="h-4 w-4" />
              <span>No active question selected</span>
            </div>
          )}
        </div>
      </div>

      <div className="flex-1 overflow-y-auto px-4 pb-6">
        <SectionTitle>Decision</SectionTitle>
        <div className="space-y-1">
          {decision.map((item) => (
            <SidebarButton
              key={item.key}
              item={item}
              activeItem={activeItem}
              onNavigate={onNavigate}
            />
          ))}
        </div>

        <SectionTitle>Evidence</SectionTitle>
        <div className="space-y-1">
          {evidence.map((item) => (
            <SidebarButton
              key={item.key}
              item={item}
              activeItem={activeItem}
              onNavigate={onNavigate}
            />
          ))}
        </div>

        <SectionTitle>Prediction</SectionTitle>
        <div className="space-y-1">
          {prediction.map((item) => (
            <SidebarButton
              key={item.key}
              item={item}
              activeItem={activeItem}
              onNavigate={onNavigate}
            />
          ))}
        </div>

        <SectionTitle>Learning</SectionTitle>
        <div className="space-y-1">
          {learning.map((item) => (
            <SidebarButton
              key={item.key}
              item={item}
              activeItem={activeItem}
              onNavigate={onNavigate}
            />
          ))}
        </div>
      </div>
    </aside>
  );
}
