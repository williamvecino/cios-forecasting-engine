// pages/ForecastCasePage.tsx

import React, { useState } from "react";
import ActorEnvironmentPanel from "../components/ActorEnvironmentPanel";
import type { ActorEnvironmentConfig } from "../lib/forecastEnvironment";
import { runForecastWithEnvironment } from "../lib/runForecast";

export default function ForecastCasePage() {
  const [environment, setEnvironment] = useState<ActorEnvironmentConfig>({
    specialtyActorProfile: "general",
    payerEnvironment: "balanced",
    guidelineLeverage: "medium",
    competitiveLandscape: "entrenched_standard_of_care",
    accessFrictionIndex: 0.5,
    adoptionPhase: "early_adoption",
    forecastHorizonMonths: 12,
  });

  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<any>(null);

  const initializeCase = async () => {
    setLoading(true);
    try {
      const output = await runForecastWithEnvironment(environment);
      setResult(output);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-[#08111f] p-6 text-white">
      <div className="mx-auto max-w-6xl space-y-6">
        <ActorEnvironmentPanel
          value={environment}
          onChange={setEnvironment}
          onInitialize={initializeCase}
          onCancel={() => setResult(null)}
        />

        <div className="rounded-3xl border border-slate-800 bg-[#0b1220] p-6">
          <div className="mb-4 text-lg font-semibold">Forecast Output</div>

          {loading && (
            <div className="text-slate-400">Running forecast...</div>
          )}

          {!loading && result && (
            <div className="space-y-5">
              <div className="grid grid-cols-1 gap-4 md:grid-cols-3">
                <MetricCard
                  label="Base Probability"
                  value={`${(result.baseProbability * 100).toFixed(1)}%`}
                />
                <MetricCard
                  label="Posterior Probability"
                  value={`${(result.posteriorProbability * 100).toFixed(1)}%`}
                />
                <MetricCard
                  label="Adjusted Probability"
                  value={`${(result.adjustedProbability * 100).toFixed(1)}%`}
                />
              </div>

              <div className="rounded-2xl border border-slate-800 bg-slate-950/40 p-4">
                <div className="mb-2 font-medium text-slate-200">
                  Environment Adjustments
                </div>
                <div className="text-sm text-slate-400">
                  Prior multiplier: {result.environmentAdjustments.priorMultiplier}
                </div>
                <div className="text-sm text-slate-400">
                  Posterior multiplier: {result.environmentAdjustments.posteriorMultiplier}
                </div>

                <ul className="mt-3 space-y-2 text-sm text-slate-400">
                  {result.environmentAdjustments.explanation.map(
                    (line: string, idx: number) => (
                      <li key={idx}>• {line}</li>
                    )
                  )}
                </ul>
              </div>

              <div className="rounded-2xl border border-slate-800 bg-slate-950/40 p-4">
                <div className="mb-2 font-medium text-slate-200">Key Drivers</div>
                <ul className="space-y-2 text-sm text-slate-400">
                  {result.keyDrivers.map((driver: string, idx: number) => (
                    <li key={idx}>• {driver}</li>
                  ))}
                </ul>
              </div>
            </div>
          )}

          {!loading && !result && (
            <div className="text-slate-500">
              Configure the environment and initialize the case.
            </div>
          )}
        </div>
      </div>
    </div>
  );
}

function MetricCard({ label, value }: { label: string; value: string }) {
  return (
    <div className="rounded-2xl border border-slate-800 bg-slate-950/40 p-4">
      <div className="text-sm text-slate-400">{label}</div>
      <div className="mt-2 text-2xl font-semibold text-white">{value}</div>
    </div>
  );
}
